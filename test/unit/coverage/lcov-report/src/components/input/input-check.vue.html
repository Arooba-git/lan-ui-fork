<div class="hx-box">

      ref="input"
      v-model="inputValue"
      v-bind="$props"
      :type="_type"
      :disabled="disabled"
      :readonly="readonly"
      :autocomplete="autocomplete"
      :autofocus="autofocus"
      :phone="phone"
      @focus="handleFocus"
      @blur="handleBlur"
      @change="changeHander"

        <i :class="eyeClass"></i>

<script type="text/ecmascript-6">
  import inputMixin from '../../common/mixins/input'
  const COMPONENT_NAME = 'hx-input'
  const EVENT_INPUT = 'input'
  const EVENT_BLUR = 'blur'
  const EVENT_FOCUS = 'focus'
 
  export default {
    name: COMPONENT_NAME,
    mixins: [inputMixin],
    props: {
      value: [String, Number],
      type: {
        type: String,
        default: 'text'
      },
      disabled: {
        type: Boolean,
        default: false
      },
      // placeholder: String,
      readonly: {
        type: Boolean,
        default: false
      },
      phone: {
        type: Boolean,
        default: false
      },
      autofocus: {
        type: Boolean,
        default: false
      },
      autocomplete: {
        type: [Boolean, String],
        default: false
      },
      name: String,
      id: String,
      form: String,
      minlength: Number,
      maxlength: Number,
      resize: String,
      min: Number,
      max: Number,
      step: Number,
      tabindex: String,
      clearable: {
        type: [Boolean, Object],
        default: false
      },
      eye: {
        type: [Boolean, Object],
        default: false
      }
    },
    data() {
      return {
        inputValue: this.value,
        isFocus: false,
        formatedClearable: {
          visible: false,
          blurHidden: true
        },
        formatedEye: {
          open: false,
          reverse: false
        },
        phonefalse: false,
        phonetrue: false,
        title: '',
        warnMsg: '',
        eyeVal: [],
        splitVal: '',
        // xwb
        domainList: ['@qq.com', '@gmail.com', '@163.com', '@yahoo.com', '@sina.com', '@hotmail.com', '@outlook.com', '@sohu.com', '@vip.163.com', '@vip.126.com', '@188.com', '@foxmail.com', '@live.cn', '@yeah.net', '@live.com', '@tom.com', '@aliyun.com', '@21.cn', '@corp.netease.com', '@msn.com', '@letv.com'],
        domainShow: false,
        placeholder: ''
        // xwb
      }
    },
    computed: {
      /**
       * xwb
       */
      _type() {
        const type = this.type
        console.log('hx---%s', type)
        if (type === 'password' && this.eye && this.pwdVisible) {
          return 'text'
        }
        if (type === 'phone') {
          this.title = '手机号码'
          this.placeholder = '请输入手机号码'
        } else if (type === 'email') {
          this.placeholder = '请输入邮箱地址'
          this.title = '邮箱地址'
        } else if (type === 'idcard') {
          this.placeholder = '请输入身份证号'
          this.title = '身份证'
        } else {
          this.title = ''
        }
        return type
      },
      /**
       * xwb
       */
      _showClear() {
        // let visible = this.formatedClearable.visible && this.inputValue && !this.readonly && !this.disabled
        let visible = this.inputValue && !this.readonly && !this.disabled
        // console.log(this.formatedClearable.visible + '---' + this.inputValue + '---' + !this.readonly + '---' + !this.disabled)
        // let visible = true
        if (this.formatedClearable.blurHidden && !this.isFocus) {
          visible = false
        }
        return visible
      },
      _showPwdEye() {
        return this.type === 'password' && this.eye && !this.disabled
      },
      pwdVisible() {
        const eye = this.formatedEye
        return eye.reverse ? !eye.open : eye.open
      },
      eyeClass() {
        return this.formatedEye.open ? 'cubeic-eye-visible' : 'cubeic-eye-invisible'
      }
    },
    watch: {
      /**
       * xwb
       */
      inputValue(newValue, oldValue) {
        this.$emit(EVENT_INPUT, newValue)
        // 手机号校验
        if (this.type === 'phone') {
          // 手机号码344格式
          if (newValue > oldValue) {
            if (newValue.length > 3) {
              newValue = newValue.replace(/\s*/g, '')
              if (newValue.length > 7) {
                newValue = newValue.replace(/^(...)(....)/g, '$1 $2 ')
              } else if (newValue.length > 3) {
                newValue = newValue.replace(/^(...)/g, '$1 ')
              }
            }
          } else {
            // 删除
            if (newValue.length === 9 || newValue.length === 4) {
              this.inputValue = newValue.trim()
            } else {
              this.inputValue = newValue
            }
          }
          this.inputValue = newValue.slice(0, 13)
          newValue = newValue.replace(/\s/g, '')
          var regPhone = /^1[3456789]\d{9}$/
          if (newValue.length === 11) {
            if (!(regPhone.test(newValue))) {
              this.phonefalse = true
              this.phonetrue = false
              this.warnMsg = '请输入正确的手机号码！'
            } else {
              this.phonetrue = true
              this.phonefalse = false
              this.warnMsg = ''
            }
          } else if (newValue.length > 11) {
            this.phonefalse = true
            this.phonetrue = false
            this.warnMsg = '请输入正确的手机号码！'
          } else {
            this.phonefalse = false
            this.phonetrue = false
            this.warnMsg = ''
          }
          if (newValue === '') {
            this.warnMsg = '请输入手机号码！'
          }
        }
        // 身份证校验
        if (this.type === 'idcard') {
          this.inputValue = newValue.slice(0, 18)
          var regIdCard = /^[1-9]\d{5}(18|19|20)\d{2}((0[1-9])|(1[0-2]))(([0-2][1-9])|10|20|30|31)\d{3}[0-9Xx]$/
          if (newValue.length === 18) {
            if (!(regIdCard.test(newValue))) {
              this.phonefalse = true
              this.phonetrue = false
              this.warnMsg = '您输入的身份证号码长度或者格式错误'
            } else {
              this.phonetrue = true
              this.phonefalse = false
              this.warnMsg = ''
            }
          } else if (newValue.length > 18) {
            this.phonefalse = true
            this.phonetrue = false
            this.warnMsg = '您输入的身份证号码长度或者格式错误'
          } else {
            this.phonefalse = false
            this.phonetrue = false
            this.warnMsg = ''
          }
          if (newValue === '') {
            this.warnMsg = '请输入身份证号！'
          }
        }
        if (this.type === 'email') {
          this.inputValue = newValue.toLocaleLowerCase()
          if (this.inputValue !== '' || this.inputValue !== undefined) {
            this.splitVal = this.inputValue.split('@')[0]
          }
        }
      },
      /**
       * xwb
       */
      clearable: {
        handler() {
          this.formatClearable()
        },
        deep: true,
        immediate: true
      },
      eye: {
        handler() {
          this.formateEye()
        },
        deep: true,
        immediate: true
      }
    },
    methods: {
      formatClearable() {
        if (typeof this.clearable === 'boolean') {
          this.formatedClearable.visible = this.clearable
        } else {
          Object.assign(this.formatedClearable, this.clearable)
        }
      },
      formateEye() {
        if (typeof this.eye === 'boolean') {
          this.formatedEye.open = this.eye
        } else {
          Object.assign(this.formatedEye, this.eye)
        }
      },
      handleFocus(e) {
        // console.log('聚焦')
        this.$emit(EVENT_FOCUS, e)
        this.isFocus = true
        if (this.eyeVal.length !== 0) {
          this.inputValue = this.eyeVal[0]
        }
        this.eyeVal = []
        this.formatedEye.open = true
        this.phonefalse = false
        this.phonetrue = false
        if (this.type === 'email' && this.readonly === false) {
          this.domainShow = true
        }
      },
      handleBlur(e) {
        // 失焦校验
        if (this.type === 'phone') {
          this.checkPhone()
        }
        if (this.type === 'email') {
          this.domainShow = false
          this.checkEmail()
        }
        if (this.type === 'idcard') {
          this.checkIdcard()
        }
        this.$emit(EVENT_BLUR, e)
        this.isFocus = false
      },
      handleClear(e) {
        this.inputValue = ''
        this.$refs.input.focus()
      },
      handlePwdEye() {
        this.formatedEye.open = !this.formatedEye.open
        if (this.type === 'phone') {
          this.eyeVal.push(this.inputValue)
          console.log(this.eyeVal[0])
          var str1 = this.inputValue.slice(0, 4)
          var str3 = this.inputValue.slice(8, 13)
          if (this.formatedEye.open === false) {
            this.inputValue = str1 + '****' + str3
          } else if (this.formatedEye.open === true) {
            this.inputValue = this.eyeVal[0]
          }
        }
      },
      // 失焦校验手机号
      checkPhone() {
        // debugger
        var regPhone = /^1[3456789]\d{1}\s\d{4}\s\d{4}$/
        if (this.inputValue !== '' && this.inputValue !== undefined) {
          if (this.inputValue.length === 13) {
            if (!(regPhone.test(this.inputValue))) {
              this.phonefalse = true
              this.phonetrue = false
              this.warnMsg = '请输入正确的手机号码！'
            } else {
              this.phonetrue = true
              this.phonefalse = false
              this.warnMsg = ''
            }
            // console.log(this.phonetrue + '\n' + this.phonefalse + '\n' + this.warnMsg)
          } else if (this.inputValue === '') {
            this.phonefalse = true
            this.phonetrue = false
            this.warnMsg = '请输入手机号码！'
            // console.log(this.warnMsg + '失焦')
          } else {
            this.phonefalse = true
            this.phonetrue = false
            this.warnMsg = '请输入正确的手机号码！'
            // console.log(this.phonetrue + '\n' + this.phonefalse + '\n' + this.warnMsg)
          }
        } else {
          this.phonefalse = true
          this.phonetrue = false
          this.warnMsg = '请输入手机号码！'
        }
      },
      // 失焦校验身份证号
      checkIdcard() {
        var regIdCard = /^[1-9]\d{5}(18|19|20)\d{2}((0[1-9])|(1[0-2]))(([0-2][1-9])|10|20|30|31)\d{3}[0-9Xx]$/
        // console.log(this.inputValue)
        if (this.inputValue !== '' && this.inputValue !== undefined) {
          if (this.inputValue.length === 18) {
            if (!(regIdCard.test(this.inputValue))) {
              this.phonefalse = true
              this.phonetrue = false
              this.warnMsg = '您输入的身份证号码长度或者格式错误'
            } else {
              this.phonetrue = true
              this.phonefalse = false
              this.warnMsg = ''
            }
          } else if (this.inputValue === '') {
            this.phonefalse = true
            this.phonetrue = false
            this.warnMsg = '请输入身份证号！'
          } else {
            this.phonefalse = true
            this.phonetrue = false
            this.warnMsg = '您输入的身份证号码长度或者格式错误'
          }
        } else {
          this.phonefalse = true
          this.phonetrue = false
          this.warnMsg = '请输入身份证号！'
        }
      },
      // 邮箱域名点击事件
      clickTap(val) {
        // console.log('邮箱地址点击事件', val)
        // this.inputValue = this.inputValue + val
        console.log(val)
        this.inputValue = val
        this.domainShow = false
        this.checkEmail()
        // console.log('邮箱地址点击事件', this.domainShow)
      },
      // 失焦校验邮箱
      checkEmail() {
        // console.log('失焦校验邮箱', this.domainShow)
        // var regEmail = /^([!#$%&'*+-/?^{|}~a-z0-9\u4E00-\u9FA5\uFF00-\uFFFF]+[.]?)*[!#$%&'*+-/?^{|}~a-z0-9\u4E00-\u9FA5\uFF00-\uFFFF]+@([a-z0-9\u4E00-\u9FA5]+[-._~]?)*[a-z0-9\u4E00-\u9FA5]+\.[a-z]{2,6}$/
        var regEmail = /^([a-z0-9\u4E00-\u9FA5\uFF00-\uFFFF/?^{|}~!#$%&'*+-]+[.]?)*[a-z0-9\u4E00-\u9FA5\uFF00-\uFFFF/?^{|}~!#$%&'*+-]+@([a-z0-9\u4E00-\u9FA5]+[-._~]?)*[a-z0-9\u4E00-\u9FA5]+\.[a-z]{2,6}$/
        if (!(regEmail.test(this.inputValue))) {
          this.phonefalse = true
          this.phonetrue = false
          this.warnMsg = '请输入正确的邮箱地址！'
        } else {
          this.phonefalse = false
          this.phonetrue = true
          this.warnMsg = ''
        }
        if (this.inputValue === '') {
          this.warnMsg = '请输入邮箱地址！'
        }
      }
    }
  }
</script>
