     * 手机号码装饰器

     * formatPhone('17610242070') -> '176 1024 2070'
     * formatPhone('1761024207086235') -> '176 1024 2070 8623 5'

     */
    formatPhone(phone) {
      return Array.prototype.reduce.call(phone, (temp, num, i) => temp + num + ((i % 4 === 2) ? ' ' : ''))

    inputCheckPhone() {
      if (!this.value /* 值校验不通过 */) {
        this.phoneErrorInfo = '请输入手机号码！'
